ğŸ’» Usage Examples

Export Template

import { createTemplateFromEditor } from 'canva-editor/utils/templateExport'; const templateData = createTemplateFromEditor( currentPages, 'My Template', { description: 'A beautiful template', author: 'John Doe', tags: ['business', 'modern'], download: true } );

Import Template with Parameters

import { loadTemplateFromFile, importTemplate } from 'canva-editor/utils/templateImport'; const templateData = await loadTemplateFromFile(file); const customizedPages = importTemplate(templateData, { templatePath: './template.json', textParameters: { title: 'Welcome!', subtitle: 'Get started today' }, imageParameters: { backgroundImage: 'https://example.com/bg.jpg' } });

React Components

import { TemplateManager, TemplateCustomizer } from 'canva-editor/components/template'; // Template Manager Component <TemplateManager onClose={() => setShowManager(false)} /> // Template Customizer Component <TemplateCustomizer templateData={templateData} onApply={(customizedPages) => { // Apply to editor customizedPages.forEach((page, index) => { actions.setPage(activePage + index, page); }); }} />
